数学函数 - exp

原型：extern float exp(float x);

用法：#include <math.h>

功能：求e的x次幂

说明：e=2.718281828...

举例：


      // exp.c
      
      #include <syslib.h>
      #include <math.h>

      main()
      {
        clrscr();        // clear screen
        textmode(0x00);  // 6 lines per LCD screen
        
        printf("e=%f\n",exp(1.0));
        
        getchar();
        return 0;
      }
