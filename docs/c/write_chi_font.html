标准库函数 - write_chi_font

原型：extern void write_chi_font(int x,int y,char *buf);

用法：#include <system.h>

功能：写汉字点阵

说明：可以把get_chi_font所取回的点阵信息写到屏幕上指定位置。

举例：

      // write_chi_font.c
      
      #include <system.h>

      main()
      {
        char buf[32];
        char *s="文曲星",*p;
        int x=0,y=0;   // screen location to display HZ

        clrscr();
        gotoxy(10,10); // Hide Cursor
        
        p=s;
        while(*p)
        {
          get_chi_font(p,buf);
          write_chi_font(x,y,buf);
          p+=2;        // forward a Chinese Character(HZ), two bytes
          x+=16;
        }
        
        getchar();
        return 0;
      }
