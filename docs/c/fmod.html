数学函数 - fmod

原型：extern float fmod(float x, float y);

用法：#include <math.h>

功能：计算x/y的余数

说明：返回x-n*y，符号同y。n=[x/y](向离开零的方向取整)

举例：


      // fmod.c
      
      #include <syslib.h>
      #include <math.h>

      main()
      {
        float x,y;
        
        clrscr();        // clear screen
        textmode(0x00);  // 6 lines per LCD screen

        x=74.12;
        y=6.4;
        printf("74.12/6.4: %f\n",fmod(x,y));        
        x=74.12;
        y=-6.4;
        printf("74.12/(-6.4): %f\n",fmod(x,y));        
        
        getchar();
        return 0;
      }
