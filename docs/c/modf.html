数学函数 - modf

原型：extern float modf(float num, float *i);

用法：#include <math.h>

功能：将浮点数num分解成整数部分和小数部分。

说明：返回小数部分，将整数部分存入*i所指内存中。

举例：

      // modf.c
      
      #include <syslib.h>
      #include <math.h>

      main()
      {
        float x, i;
        
        clrscr();        // clear screen
        textmode(0x00);  // 6 lines per LCD screen
      
        x=modf(-74.12,&i);
        printf("-74.12=%.0f+(%.2f)",i,x);
        
        getchar();
        return 0;
      }
