数学函数 - floor

原型：extern float floor(float x);

用法：#include <math.h>

功能：求不大于x的最达整数

说明：返回x的下限，如74.12的下限为74，-74.12的下限为-75。返回值为float类型。

举例：

      // floor.c
      
      #include <syslib.h>
      #include <math.h>

      main()
      {
        float x;
        
        clrscr();        // clear screen
        textmode(0x00);  // 6 lines per LCD screen
        
        x=74.12;
        printf("floor(%.2f)=%.0f\n",x,floor(x));
        x=-74.12;
        printf("floor(%.2f)=%.0f\n",x,floor(x));
        
        getchar();
        return 0;
      }
