字符函数 - iscntrl

原型：extern int iscntrl(int c);

用法：#include <ctype.h>

功能：判断字符c是否为控制字符

说明：当c在0x00-0x1F之间或等于0x7F(DEL)时，返回非零值，否则返回零。

举例：

      // iscntrl.c
      
      #include <syslib.h>
      #include <ctype.h>

      main()
      {
        int c;
        
        clrscr();        // clear screen
        c='a';
        printf("%x:%s\n",c,iscntrl(c)?"yes":"no");
        c=0x0d;
        printf("%x:%s\n",c,iscntrl(c)?"yes":"no");
        c=0x7f;
        printf("%x:%s\n",c,iscntrl(c)?"yes":"no");
        getchar();
        return 0;
      }
