数学函数 - ceil

原型：extern float ceil(float x);

用法：#include <math.h>

功能：求不小于x的最小整数

说明：返回x的上限，如74.12的上限为75，-74.12的上限为-74。返回值为float类型。

举例：


      // ceil.c
      
      #include <syslib.h>
      #include <math.h>

      main()
      {
        float x;
        
        clrscr();        // clear screen
        textmode(0x00);  // 6 lines per LCD screen
        
        x=74.12;
        printf("ceil(%.2f)=%.0f\n",x,ceil(x));
        x=-74.12;
        printf("ceil(%.2f)=%.0f\n",x,ceil(x));
        
        getchar();
        return 0;
      }
