字符串函数 - bcopy

原型：extern void bcopy(const void *src, void *dest, int n);

用法：#include <string.h>

功能：将字符串src的前n个字节复制到dest中

说明：bcopy不检查字符串中的空字节NULL，函数没有返回值。

举例：

      // bcopy.c
      
      #include <syslib.h>
      #include <string.h>

      main()
      {
        char *s="Golden Global View";
        char d[20];
        
        clrscr();        // clear screen
        bcopy(s,d,6);
        printf("s: %s\n",s);
        printf("d: %s\n",d);

        getchar();
        clrscr();
        s[13]=0;
        bcopy(s+7,d,11);  // bcopy ignore null in string
        printf("%s\n",s+7);
        for(i=0;i<11;i++)
          putchar(d[i]);

        getchar();
        return 0;
      }
