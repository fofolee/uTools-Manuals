func IsSurrogate(r rune) bool

func DecodeRune 

func DecodeRune(r1, r2 rune) rune

func Decode(s []uint16) []rune

将utf-16序列解码为unicode码值序列。

func EncodeRune 

func EncodeRune(r rune) (r1, r2 rune)

func Encode(s []rune) []uint16
