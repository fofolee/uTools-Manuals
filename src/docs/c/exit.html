标准库函数 - exit

原型：extern void exit(int retval);

用法：#include <stdlib.h>

功能：结束程序

说明：返回值将被忽略

举例：


      // exit.c
      
      #include <syslib.h>
      #include <stdlib.h>

      main()
      {
        int i;
        
        clrscr();        // clear screen
        textmode(0x00);
        
        for(i=0;i<10;i++)
        {
          if(i==5) exit(0);
          else
          {
            clrscr();
            printf("%d",i);
            getchar();
          }
        }
        
        getchar();
        return 0;
      }
